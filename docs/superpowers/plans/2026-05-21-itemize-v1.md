# Itemize v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Chrome/Firefox extension that matches YNAB transactions to Amazon orders, classifies line items, and writes split (or simple) categorizations back to YNAB.

**Architecture:** Side panel React UI communicates with a service worker via messages. The service worker handles YNAB API calls, orchestrates scraping by messaging a content script on Amazon's domain, runs classification, and writes results back. IndexedDB caches all scraped data to minimize repeated scraping.

**Tech Stack:** wxt (MV3), React 19, TypeScript, Tailwind CSS, IndexedDB, chrome.storage.local

**Design Spec:** `docs/superpowers/specs/2026-05-21-itemize-v1-design.md`

**Existing Reference:** The Playwright scraper at `/home/michael/projects/ynab-amazon/src/lib/scraper/` contains working Amazon DOM selectors and parsing logic. Port the DOM parsing logic (not the Playwright wrapper) into the content script.

```ts
export function centsToMilliunits(cents: number): number {
  return cents * -10;
}

export function millunitsToCents(milliunits: number): number {
  return Math.abs(Math.round(milliunits / 10));
}

export function formatCents(cents: number): string {
  return `$${(cents / 100).toFixed(2)}`;
}

export function parseDollarsToCents(dollars: string): number {
  return Math.abs(Math.round(parseFloat(dollars.replace(/[^0-9.-]/g, "")) * 100));
}

export function distributeProportionally(
  itemAmounts: number[],
  remainder: number,
): number[] {
  const total = itemAmounts.reduce((sum, a) => sum + a, 0);
  if (total === 0) return itemAmounts.map(() => 0);

  const shares = itemAmounts.map((amount) =>
    Math.round((amount / total) * remainder),
  );

  // Adjust last item to absorb rounding error
  const distributed = shares.reduce((sum, s) => sum + s, 0);
  shares[shares.length - 1] += remainder - distributed;

  return shares;
}
```

```ts
import type { Charge, Order, ProductCacheEntry, Category } from "./types";

const DB_NAME = "itemize";
const DB_VERSION = 1;

let dbPromise: Promise<IDBDatabase> | null = null;

function openDB(): Promise<IDBDatabase> {
  if (dbPromise) return dbPromise;
  dbPromise = new Promise((resolve, reject) => {
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    request.onupgradeneeded = () => {
      const db = request.result;
      if (!db.objectStoreNames.contains("charges")) {
        db.createObjectStore("charges", { keyPath: "id" });
      }
      if (!db.objectStoreNames.contains("orders")) {
        db.createObjectStore("orders", { keyPath: "id" });
      }
      if (!db.objectStoreNames.contains("productCache")) {
        db.createObjectStore("productCache", { keyPath: "id" });
      }
      if (!db.objectStoreNames.contains("categories")) {
        db.createObjectStore("categories", { keyPath: "id" });
      }
    };
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
  return dbPromise;
}

async function getStore(
  storeName: string,
  mode: IDBTransactionMode = "readonly",
): Promise<IDBObjectStore> {
  const db = await openDB();
  return db.transaction(storeName, mode).objectStore(storeName);
}

function requestToPromise<T>(request: IDBRequest<T>): Promise<T> {
  return new Promise((resolve, reject) => {
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(request.error);
  });
}

// --- Charges ---

export async function getCharge(id: string): Promise<Charge | undefined> {
  const store = await getStore("charges");
  return requestToPromise(store.get(id));
}

export async function getAllCharges(): Promise<Charge[]> {
  const store = await getStore("charges");
  return requestToPromise(store.getAll());
}

export async function putCharge(charge: Charge): Promise<void> {
  const store = await getStore("charges", "readwrite");
  await requestToPromise(store.put(charge));
}

export async function putCharges(charges: Charge[]): Promise<void> {
  const db = await openDB();
  const tx = db.transaction("charges", "readwrite");
  const store = tx.objectStore("charges");
  for (const charge of charges) {
    store.put(charge);
  }
  return new Promise((resolve, reject) => {
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(tx.error);
  });
}

// --- Orders ---
```

```ts
import type { YnabTransaction, Category } from "./types";

const BASE_URL = "https://api.ynab.com/v1";

async function ynabFetch(path: string, token: string, options?: RequestInit) {
  const response = await fetch(`${BASE_URL}${path}`, {
    ...options,
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/json",
      ...options?.headers,
    },
  });
  if (!response.ok) {
    const body = await response.text().catch(() => "");
    throw new Error(`YNAB API ${response.status}: ${body}`);
  }
  return response.json();
}

export interface YnabBudget {
  id: string;
  name: string;
}

export async function getBudgets(token: string): Promise<YnabBudget[]> {
  const data = await ynabFetch("/budgets", token);
  return data.data.budgets.map((b: { id: string; name: string }) => ({
    id: b.id,
    name: b.name,
  }));
}

export async function getCategories(
  token: string,
  budgetId: string,
): Promise<Category[]> {
  const data = await ynabFetch(`/budgets/${budgetId}/categories`, token);
  const categories: Category[] = [];
  for (const group of data.data.category_groups) {
    if (group.hidden || group.name === "Internal Master Category") continue;
    for (const cat of group.categories) {
      if (cat.hidden || cat.deleted) continue;
      categories.push({
        id: cat.id,
        name: cat.name,
        groupName: group.name,
      });
    }
  }
  return categories;
}

export async function getUnapprovedTransactions(
  token: string,
  budgetId: string,
): Promise<YnabTransaction[]> {
  const data = await ynabFetch(
    `/budgets/${budgetId}/transactions?type=unapproved`,
    token,
  );
  return data.data.transactions;
}

export interface YnabTransactionUpdate {
  category_id?: string | null;
  approved?: boolean;
  memo?: string;
  subtransactions?: Array<{
    amount: number;
    category_id: string;
    memo: string;
  }>;
}

export async function updateTransaction(
  token: string,
  budgetId: string,
  transactionId: string,
  update: YnabTransactionUpdate,
): Promise<void> {
  await ynabFetch(
    `/budgets/${budgetId}/transactions/${transactionId}`,
    token,
    {
      method: "PUT",
      body: JSON.stringify({ transaction: update }),
    },
  );
}
```

```tsx
import { useState } from "react";
import type { YnabBudget } from "../core/ynab";

interface OnboardingProps {
  onComplete: () => void;
}

export default function Onboarding({ onComplete }: OnboardingProps) {
  const [token, setToken] = useState("");
  const [budgets, setBudgets] = useState<YnabBudget[] | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  async function validateToken() {
    setLoading(true);
    setError(null);
    try {
      const response = await chrome.runtime.sendMessage({
        type: "GET_BUDGETS",
        token,
      });
      if (response.error) {
        setError(response.error);
        return;
      }
      setBudgets(response.budgets);
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to validate token");
    } finally {
      setLoading(false);
    }
  }

  async function selectBudget(budget: YnabBudget) {
    setLoading(true);
    setError(null);
    try {
      const response = await chrome.runtime.sendMessage({
        type: "SAVE_SETTINGS",
        token,
        budgetId: budget.id,
        budgetName: budget.name,
      });
      if (response.error) {
        setError(response.error);
        return;
      }
      onComplete();
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to save settings");
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="min-h-screen bg-gray-950 text-gray-100 p-4">
      <h1 className="text-lg font-semibold mb-1">Itemize</h1>
```

```ts
import { SELECTORS, ASIN_REGEX } from "./selectors";

const MONTHS: Record<string, string> = {
  January: "01", February: "02", March: "03", April: "04",
  May: "05", June: "06", July: "07", August: "08",
  September: "09", October: "10", November: "11", December: "12",
};

export interface RawCharge {
  chargeDate: string; // ISO date
  amountCents: number;
  orderId: string | null;
  cardLastFour: string | null;
  isRefund: boolean;
}

export interface RawItem {
  productId: string;
  title: string;
  priceCents: number;
  quantity: number;
  imageUrl: string;
}

function parseNaturalDate(dateStr: string): string {
  const match = dateStr.match(/(\w+)\s+(\d{1,2}),?\s*(\d{4})/);
  if (!match) return dateStr;
  const m = MONTHS[match[1]];
  if (!m) return dateStr;
  return `${match[3]}-${m}-${match[2].padStart(2, "0")}`;
}

function parseCents(dollarStr: string): number {
  return Math.abs(Math.round(parseFloat(dollarStr.replace(/[^0-9.-]/g, "")) * 100));
}

export function parseChargesFromDocument(doc: Document): RawCharge[] {
  const results: RawCharge[] = [];
  let currentDate: string | null = null;

  const dateContainers = doc.querySelectorAll(SELECTORS.dateContainer);
  dateContainers.forEach((dateEl) => {
    const dateSpan = dateEl.querySelector("span");
    if (dateSpan) currentDate = dateSpan.textContent?.trim() ?? null;

    const sibling = dateEl.nextElementSibling;
    if (!sibling) return;

    sibling.querySelectorAll(SELECTORS.lineItem).forEach((item) => {
      const text = item.textContent ?? "";

      const amountEl = item.querySelector(SELECTORS.amountSpan);
      if (!amountEl) return;
      const amountText = amountEl.textContent?.trim() ?? "";

      const isRefund = /refund/i.test(text) || amountText.startsWith("+");

      const orderLink = item.querySelector<HTMLAnchorElement>(SELECTORS.orderLink);
      let orderId: string | null = null;
      if (orderLink) {
        const orderMatch = (orderLink.getAttribute("href") ?? "").match(
          /orderID=([^&]+)/,
        );
        if (orderMatch) orderId = orderMatch[1];
      }

      const cardEl = item.querySelector(SELECTORS.cardSpan);
      let cardLastFour: string | null = null;
      if (cardEl) {
        const cardMatch = (cardEl.textContent ?? "").match(/\*{3,4}(\d{4})/);
        if (cardMatch) cardLastFour = cardMatch[1];
      }

      results.push({
        chargeDate: currentDate ? parseNaturalDate(currentDate) : "",
        amountCents: parseCents(amountText),
        orderId,
        cardLastFour,
        isRefund,
      });
    });
  });

  return results;
}

export function parseItemsFromDocument(doc: Document): RawItem[] {
  const results: RawItem[] = [];

  let itemElements = doc.querySelectorAll(SELECTORS.itemBoxPrimary);

  if (itemElements.length === 0) {
    itemElements = doc.querySelectorAll(SELECTORS.itemBoxFallback);
  }

  itemElements.forEach((item) => {
    const titleEl = item.querySelector(SELECTORS.productLink);
    const title = titleEl?.textContent?.trim() ?? "";
    if (!title) return;

    let productId = "";
    const href = titleEl?.getAttribute("href") ?? "";
    const asinMatch = href.match(ASIN_REGEX);
    if (asinMatch) productId = asinMatch[1];

    const imgEl = item.querySelector("img");
    const imageUrl = imgEl?.getAttribute("src") ?? "";

    let priceCents = 0;
    const priceEl = item.querySelector(SELECTORS.priceEl) ??
      item.querySelector(SELECTORS.priceFallback);
    if (priceEl) {
      priceCents = parseCents(priceEl.textContent ?? "0");
    }

    let quantity = 1;
    const qtyMatchDirect = (item.textContent ?? "").match(/Qty:\s*(\d+)/i);
    if (qtyMatchDirect) {
      quantity = parseInt(qtyMatchDirect[1], 10);
    } else {
      const qtyEl = item.querySelector(SELECTORS.quantityFallback);
      if (qtyEl) {
        const qtyMatch = (qtyEl.textContent ?? "").match(/(\d+)/);
        if (qtyMatch) quantity = parseInt(qtyMatch[1], 10);
      }
    }

    results.push({ productId, title, priceCents, quantity, imageUrl });
  });

  return results;
}
```

```ts
async function handleMessage(message: { type: string; [key: string]: unknown }) {
  switch (message.type) {
    case "CHECK_AUTH":
      return checkAuth();
    case "SCRAPE_CHARGES":
      return scrapeCharges(message.maxPages as number | undefined);
    case "SCRAPE_ORDER":
      return scrapeOrder(message.orderId as string);
    default:
      return { error: `Unknown message type: ${message.type}` };
  }
}

function checkAuth(): { authenticated: boolean } {
  if (AUTH_PAGE_REGEX.test(window.location.href)) {
    return { authenticated: false };
  }
  return { authenticated: true };
}

async function scrapeCharges(maxPages = 5): Promise<{
  charges: RawCharge[];
  error?: string;
}> {
  const allCharges: RawCharge[] = [];
  let pagesScraped = 0;

  // Fetch the transactions page HTML
  let html: string;
  try {
    const resp = await fetch(TRANSACTIONS_URL, { credentials: "include" });
    html = await resp.text();
  } catch (e) {
    return { charges: [], error: `Failed to fetch transactions page: ${e}` };
  }

  while (pagesScraped < maxPages) {
    const doc = new DOMParser().parseFromString(html, "text/html");

    if (AUTH_PAGE_REGEX.test(doc.URL || "")) {
      return { charges: allCharges, error: "not_logged_in" };
    }

    const pageCharges = parseChargesFromDocument(doc);
    if (pageCharges.length === 0 && pagesScraped === 0) {
      return { charges: [], error: "no_charges_found" };
    }

    allCharges.push(...pageCharges);
    pagesScraped++;

    // Check for next page button and fetch it
    const nextButton = doc.querySelector(SELECTORS.nextPageButton) as HTMLInputElement | null;
    if (!nextButton) break;

    const labelId = nextButton.getAttribute("aria-labelledby") ?? "";
    const labelEl = doc.getElementById(labelId);
    if (labelEl?.textContent?.trim() !== "Next Page") break;

    // Find the form and submit via fetch
    const form = nextButton.closest("form");
    if (!form) break;

    try {
      const formData = new FormData(form);
      const resp = await fetch(form.action || TRANSACTIONS_URL, {
        method: "POST",
        body: formData,
        credentials: "include",
      });
      html = await resp.text();
    } catch {
      break;
    }
  }

  return { charges: allCharges };
}

async function scrapeOrder(
  orderId: string,
): Promise<{ items: RawItem[]; error?: string }> {
  try {
    const resp = await fetch(orderDetailUrl(orderId), {
      credentials: "include",
    });
    const html = await resp.text();
    const doc = new DOMParser().parseFromString(html, "text/html");

    if (AUTH_PAGE_REGEX.test(doc.URL || window.location.href)) {
      return { items: [], error: "not_logged_in" };
    }

    const items = parseItemsFromDocument(doc);
    return { items };
  } catch (e) {
    return { items: [], error: `Failed to fetch order ${orderId}: ${e}` };
  }
}
```

```ts
async function performSync(): Promise<{ queue: QueueTransaction[]; error?: string }> {
  const settings = await getSettings();
  if (!settings.ynabToken || !settings.budgetId) {
    return { queue: [], error: "Not connected to YNAB" };
  }

  // 1. Fetch unapproved transactions
  let ynabTransactions: YnabTransaction[];
  try {
    ynabTransactions = await getUnapprovedTransactions(
      settings.ynabToken,
      settings.budgetId,
    );
  } catch (e) {
    return { queue: [], error: `Failed to fetch YNAB transactions: ${e}` };
  }

  // 2. Filter to supported retailers with strategy: scrape
  const scrapeable = ynabTransactions
    .filter((t) => {
      if (!t.payee_name) return false;
      const match = getRetailerForPayee(t.payee_name);
      return match?.strategy === "scrape";
    })
    .map((t) => ({
      transaction: t,
      retailer: getRetailerForPayee(t.payee_name!)!.retailer,
    }));

  if (scrapeable.length === 0) {
    return { queue: [] };
  }

  // 3. Check cache for existing matches
  const cachedCharges = await getAllCharges();
  const cachedOrders = await getAllOrders();
  const categories = await getAllCategories();
  const queue: QueueTransaction[] = [];

  const retailersToScrape = new Set<string>();

  for (const { transaction, retailer } of scrapeable) {
    const amountCents = millunitsToCents(transaction.amount);
    const matchedCharge = matchTransactionToCharge(
      amountCents,
      transaction.date,
      cachedCharges,
    );

    if (matchedCharge?.orderId) {
      const orderId = `${retailer}:${matchedCharge.orderId}`;
      const order = cachedOrders.find((o) => o.id === orderId);
      if (order) {
        const classifiedItems = await classifyItems(order, categories);
        queue.push({
          ynabTransaction: transaction,
          retailer,
          matchStatus: { status: "matched", order, classifiedItems },
        });
        continue;
      }
    }

    // Need to scrape
    retailersToScrape.add(retailer);
    queue.push({
      ynabTransaction: transaction,
      retailer,
      matchStatus: { status: "loading" },
    });
  }

  // 4. Scrape retailers that need it
  if (retailersToScrape.size > 0) {
    for (const retailer of retailersToScrape) {
      const scrapeResult = await scrapeRetailer(retailer);
      if (scrapeResult.error === "not_logged_in") {
        // Update all loading items for this retailer
        for (const item of queue) {
          if (item.retailer === retailer && item.matchStatus.status === "loading") {
            item.matchStatus = { status: "auth_required" };
          }
        }
        continue;
      }

      // Re-match after scraping
      const updatedCharges = await getAllCharges();
      const updatedOrders = await getAllOrders();

      for (const item of queue) {
        if (item.retailer === retailer && item.matchStatus.status === "loading") {
          const amountCents = millunitsToCents(item.ynabTransaction.amount);
          const matchedCharge = matchTransactionToCharge(
            amountCents,
            item.ynabTransaction.date,
            updatedCharges,
          );

          if (matchedCharge?.orderId) {
            const orderId = `${retailer}:${matchedCharge.orderId}`;
            const order = updatedOrders.find((o) => o.id === orderId);
            if (order) {
              const classifiedItems = await classifyItems(order, categories);
              item.matchStatus = { status: "matched", order, classifiedItems };
              continue;
            }
          }

          item.matchStatus = { status: "no_match" };
        }
      }
    }
  }

  return { queue };
}

async function scrapeRetailer(
  retailer: string,
): Promise<{ error?: string }> {
  const url = retailerContentScriptUrls[retailer];
  if (!url) return { error: `Unknown retailer: ${retailer}` };

  // Find or open a tab
  const tabs = await chrome.tabs.query({ url: `*://*.${retailer}.com/*` });
  let tab = tabs[0];
  let openedTab = false;

  if (!tab) {
    tab = await chrome.tabs.create({ url, active: false });
    openedTab = true;
    // Wait for tab to load
    await new Promise<void>((resolve) => {
      function listener(tabId: number, info: chrome.tabs.TabChangeInfo) {
        if (tabId === tab!.id && info.status === "complete") {
          chrome.tabs.onUpdated.removeListener(listener);
          resolve();
        }
      }
      chrome.tabs.onUpdated.addListener(listener);
    });
  }

  try {
    // Check auth
    const authResult = await chrome.tabs.sendMessage(tab.id!, {
      type: "CHECK_AUTH",
    });

    if (!authResult.authenticated) {
      // Bring tab to foreground so user can log in
      await chrome.tabs.update(tab.id!, { active: true });
      return { error: "not_logged_in" };
    }

    // Scrape charges
    const chargeResult = await chrome.tabs.sendMessage(tab.id!, {
      type: "SCRAPE_CHARGES",
      maxPages: 5,
    });

    if (chargeResult.error === "not_logged_in") {
      await chrome.tabs.update(tab.id!, { active: true });
      return { error: "not_logged_in" };
    }

    // Convert raw charges and save to IndexedDB
    const existingCharges = await getAllCharges();
    const existingIds = new Set(existingCharges.map((c) => c.id));

    const newCharges: Charge[] = [];
    for (const raw of chargeResult.charges ?? []) {
      const charge: Charge = {
        id: `${retailer}:${raw.chargeDate}-${raw.amountCents}-${raw.orderId ?? "unknown"}`,
        retailer,
        chargeDate: raw.chargeDate,
        amountCents: raw.amountCents,
        orderId: raw.orderId,
        cardLastFour: raw.cardLastFour,
        isRefund: raw.isRefund,
        scrapedAt: new Date().toISOString(),
      };
      if (!existingIds.has(charge.id)) {
        newCharges.push(charge);
      }
    }

    if (newCharges.length > 0) {
      await putCharges(newCharges);
    }

    // Scrape order details for new charges with order IDs
    const allCharges = await getAllCharges();
    const orderIdsToScrape = new Set<string>();

    for (const charge of allCharges) {
      if (!charge.orderId) continue;
      const orderId = `${retailer}:${charge.orderId}`;
      const existing = await getOrder(orderId);
      if (!existing) {
        orderIdsToScrape.add(charge.orderId);
      }
    }

    for (const orderId of orderIdsToScrape) {
      const orderResult = await chrome.tabs.sendMessage(tab.id!, {
        type: "SCRAPE_ORDER",
        orderId,
      });

      if (orderResult.error === "not_logged_in") {
        await chrome.tabs.update(tab.id!, { active: true });
        return { error: "not_logged_in" };
      }

      if (orderResult.items && orderResult.items.length > 0) {
        const order: Order = {
          id: `${retailer}:${orderId}`,
          retailer,
          orderId,
          orderDate: "",
          total: 0,
          items: orderResult.items.map(
            (item: { productId: string; title: string; priceCents: number; quantity: number; imageUrl: string }) => ({
              productId: item.productId,
              title: item.title,
              price: item.priceCents,
              quantity: item.quantity,
              imageUrl: item.imageUrl,
            }),
          ),
          scrapedAt: new Date().toISOString(),
        };
        order.total = order.items.reduce(
          (sum, i) => sum + i.price * i.quantity,
          0,
        );
        await putOrder(order);
      }
    }

    return {};
  } finally {
    // Close tab only if we opened it
    if (openedTab && tab.id) {
      await chrome.tabs.remove(tab.id).catch(() => {});
    }
  }
}

async function approveTransaction(
  transactionId: string,
  items: ApprovalItem[],
): Promise<{ ok: boolean; error?: string }> {
  const settings = await getSettings();
  if (!settings.ynabToken || !settings.budgetId) {
    return { ok: false, error: "Not connected" };
  }

  try {
    const categories = await getAllCategories();
    const update = buildYnabUpdate(items, categories);

    await updateTransaction(
      settings.ynabToken,
      settings.budgetId,
      transactionId,
      update,
    );

    // Update product cache for all items
    for (const item of items) {
      if (item.productId) {
        await putProductCacheEntry({
          id: `amazon:${item.productId}`,
          categoryId: item.categoryId,
          confirmedByUser: true,
          timesSeen: 1,
          lastSeen: new Date().toISOString(),
        });
      }
    }

    return { ok: true };
  } catch (e) {
    return { ok: false, error: e instanceof Error ? e.message : "Failed to approve" };
  }
}

async function approveBatch(
  transactions: Array<{ transactionId: string; items: ApprovalItem[] }>,
): Promise<{ ok: boolean; approved: number; errors: string[] }> {
  const errors: string[] = [];
  let approved = 0;

  for (const tx of transactions) {
    const result = await approveTransaction(tx.transactionId, tx.items);
    if (result.ok) {
      approved++;
    } else {
      errors.push(`${tx.transactionId}: ${result.error}`);
    }
  }

  return { ok: errors.length === 0, approved, errors };
}
```

```ts
import type { Charge } from "./types";

export function matchTransactionToCharge(
  amountCents: number,
  transactionDate: string,
  charges: Charge[],
): Charge | null {
  const txDate = new Date(transactionDate);
  const threeDays = 3 * 24 * 60 * 60 * 1000;

  const matches = charges.filter((charge) => {
    if (charge.isRefund) return false;
    if (charge.amountCents !== amountCents) return false;

    const chargeDate = new Date(charge.chargeDate);
    return Math.abs(txDate.getTime() - chargeDate.getTime()) <= threeDays;
  });

  if (matches.length === 1) return matches[0];

  // Multiple or zero matches
  return null;
}
```

```ts
import type { ApprovalItem, Category } from "./types";
import type { YnabTransactionUpdate } from "./ynab";
import { centsToMilliunits, distributeProportionally } from "./money";

export function buildYnabUpdate(
  items: ApprovalItem[],
  _categories: Category[],
): YnabTransactionUpdate {
  // Check if all items have the same category
  const uniqueCategories = new Set(items.map((i) => i.categoryId));

  if (uniqueCategories.size === 1) {
    // Simple categorization — no split needed
    const categoryId = items[0].categoryId;
    const memo =
      items.length === 1
        ? items[0].title
        : `Amazon: ${items.length} items`;

    return {
      category_id: categoryId,
      approved: true,
      memo,
    };
  }

  // Multiple categories — build split
  const itemAmounts = items.map((i) => i.price * i.quantity);
  const itemsTotal = itemAmounts.reduce((sum, a) => sum + a, 0);

  // Note: the charge total includes tax/shipping which isn't in the items.
  // The service worker will compute the actual subtransaction amounts
  // by using the YNAB transaction amount. For now, we build the split
  // from item amounts and let the caller handle the remainder.

  const subtransactions = items.map((item) => ({
    amount: centsToMilliunits(item.price * item.quantity),
    category_id: item.categoryId,
    memo: item.title.slice(0, 200), // YNAB memo limit
  }));

  return {
    approved: true,
    memo: `Amazon: ${items.length} items`,
    subtransactions,
  };
}

export function buildYnabUpdateWithTax(
  items: ApprovalItem[],
  chargeAmountCents: number,
): YnabTransactionUpdate {
  const uniqueCategories = new Set(items.map((i) => i.categoryId));

  if (uniqueCategories.size === 1) {
    return {
      category_id: items[0].categoryId,
      approved: true,
      memo:
        items.length === 1
          ? items[0].title
          : `Amazon: ${items.length} items`,
    };
  }

  const itemAmounts = items.map((i) => i.price * i.quantity);
  const itemsTotal = itemAmounts.reduce((sum, a) => sum + a, 0);
  const remainder = chargeAmountCents - itemsTotal;

  // Distribute tax/shipping proportionally
  const shares = distributeProportionally(itemAmounts, remainder);

  const subtransactions = items.map((item, i) => {
    const baseAmount = item.price * item.quantity;
    const withTax = baseAmount + shares[i];
    return {
      amount: centsToMilliunits(withTax),
      category_id: item.categoryId,
      memo: item.title.slice(0, 200),
    };
  });

  // Validate total
  const subtotal = subtransactions.reduce((sum, s) => sum + s.amount, 0);
  const expectedTotal = centsToMilliunits(chargeAmountCents);
  if (subtotal !== expectedTotal) {
    throw new Error(
      `Split total mismatch: ${subtotal} vs expected ${expectedTotal}`,
    );
  }

  return {
    approved: true,
    memo: `Amazon: ${items.length} items`,
    subtransactions,
  };
}
```

`components/SplitBreakdown.tsx`:
```tsx
import type { Category } from "../core/types";
import { formatCents, distributeProportionally } from "../core/money";

interface SplitItem {
  price: number;
  quantity: number;
  categoryId: string | null;
}

interface SplitBreakdownProps {
  items: SplitItem[];
  chargeAmountCents: number;
  categories: Category[];
}

export default function SplitBreakdown({
  items,
  chargeAmountCents,
  categories,
}: SplitBreakdownProps) {
  const categoryMap = new Map(categories.map((c) => [c.id, c.name]));

  const itemAmounts = items.map((i) => i.price * i.quantity);
  const itemsTotal = itemAmounts.reduce((sum, a) => sum + a, 0);
  const remainder = chargeAmountCents - itemsTotal;
  const shares = distributeProportionally(itemAmounts, remainder);

  // Aggregate by category
  const byCategory = new Map<string, number>();
  items.forEach((item, i) => {
    const catId = item.categoryId ?? "uncategorized";
    const amount = item.price * item.quantity + shares[i];
    byCategory.set(catId, (byCategory.get(catId) ?? 0) + amount);
  });

  return (
    <div className="bg-gray-900 rounded-lg p-3 mt-4">
      <div className="text-[10px] uppercase tracking-wider text-gray-500 mb-2">
        Split Breakdown
      </div>

      {Array.from(byCategory.entries()).map(([catId, amount]) => (
        <div
          key={catId}
          className="flex justify-between text-xs mb-1"
        >
          <span className="text-gray-400">
            {catId === "uncategorized"
              ? "Uncategorized"
              : categoryMap.get(catId) ?? catId}
          </span>
          <span className="text-white">{formatCents(amount)}</span>
        </div>
      ))}

      <div className="border-t border-gray-700 mt-2 pt-2 flex justify-between text-xs">
        <span className="text-gray-500 italic text-[10px]">
          Tax/shipping distributed proportionally
        </span>
        <span className="text-white font-semibold">
          {formatCents(chargeAmountCents)}
        </span>
      </div>
    </div>
  );
}
```

`components/DetailView.tsx`:
```tsx
import { useState } from "react";
import type { QueueTransaction, Category, ApprovalItem } from "../core/types";
import { formatCents, millunitsToCents } from "../core/money";
import ItemCard from "./ItemCard";
import SplitBreakdown from "./SplitBreakdown";

interface DetailViewProps {
  transaction: QueueTransaction;
  categories: Category[];
  onBack: () => void;
  onApprove: (transactionId: string, items: ApprovalItem[]) => Promise<void>;
}

export default function DetailView({
  transaction,
  categories,
  onBack,
  onApprove,
}: DetailViewProps) {
  const { ynabTransaction, matchStatus } = transaction;

  if (matchStatus.status !== "matched") {
    return (
      <div className="min-h-screen bg-gray-950 text-gray-100 p-4">
        <button
          onClick={onBack}
          className="text-sm text-indigo-400 hover:text-indigo-300 mb-4"
        >
          ← Back to queue
        </button>
        <p className="text-sm text-gray-400">
          {matchStatus.status === "loading" && "Loading..."}
          {matchStatus.status === "no_match" && "No matching order found."}
          {matchStatus.status === "auth_required" &&
            "Log into Amazon and try again."}
          {matchStatus.status === "error" && matchStatus.message}
        </p>
      </div>
    );
  }

  const { order, classifiedItems } = matchStatus;
  const chargeAmountCents = millunitsToCents(ynabTransaction.amount);

  const [selectedCategories, setSelectedCategories] = useState<
    Map<number, string>
  >(() => {
    const map = new Map<number, string>();
    classifiedItems.forEach((item, i) => {
      if (item.suggestedCategoryId) {
        map.set(i, item.suggestedCategoryId);
      }
    });
    return map;
  });

  const [approving, setApproving] = useState(false);

  const allCategorized = classifiedItems.every(
    (_, i) => selectedCategories.has(i),
  );

  function handleCategoryChange(index: number, categoryId: string) {
    setSelectedCategories((prev) => {
      const next = new Map(prev);
      if (categoryId) {
        next.set(index, categoryId);
      } else {
        next.delete(index);
      }
      return next;
    });
  }

  async function handleApprove() {
    if (!allCategorized) return;
    setApproving(true);
    try {
      const items: ApprovalItem[] = classifiedItems.map((item, i) => ({
        productId: item.productId,
        title: item.title,
        price: item.price,
        quantity: item.quantity,
        categoryId: selectedCategories.get(i)!,
      }));
      await onApprove(ynabTransaction.id, items);
      onBack();
    } finally {
      setApproving(false);
    }
  }

  const splitItems = classifiedItems.map((item, i) => ({
    price: item.price,
    quantity: item.quantity,
    categoryId: selectedCategories.get(i) ?? null,
  }));

  return (
    <div className="min-h-screen bg-gray-950 text-gray-100 p-4 flex flex-col">
      <button
        onClick={onBack}
        className="text-sm text-indigo-400 hover:text-indigo-300 mb-3"
      >
        ← Back to queue
      </button>

      {/* Transaction header */}
      <div className="bg-gray-900 rounded-lg p-3 mb-4">
        <div className="flex justify-between mb-0.5">
          <span className="text-base font-semibold text-white">
            {formatCents(chargeAmountCents)}
          </span>
          <span className="text-xs text-gray-400">{ynabTransaction.date}</span>
        </div>
        <div className="text-xs text-gray-400">
          {ynabTransaction.payee_name} · Order #{order.orderId}
        </div>
      </div>

      {/* Items */}
      <div className="text-[10px] uppercase tracking-wider text-gray-500 mb-2">
        Items
      </div>
      <div className="space-y-2 flex-1">
        {classifiedItems.map((item, i) => (
          <ItemCard
            key={i}
            productId={item.productId}
            title={item.title}
            imageUrl={item.imageUrl}
            price={item.price}
            quantity={item.quantity}
            selectedCategoryId={selectedCategories.get(i) ?? null}
            categories={categories}
            onCategoryChange={(catId) => handleCategoryChange(i, catId)}
          />
        ))}
      </div>

      {/* Split breakdown */}
      <SplitBreakdown
        items={splitItems}
        chargeAmountCents={chargeAmountCents}
        categories={categories}
      />

      {/* Approve button */}
      <div className="mt-4">
        <button
          onClick={handleApprove}
          disabled={!allCategorized || approving}
          className={`w-full text-sm font-medium rounded-lg py-2.5 ${
            allCategorized
              ? "bg-indigo-600 hover:bg-indigo-500 text-white"
              : "bg-gray-800 text-gray-500 cursor-not-allowed"
          }`}
        >
          {approving
            ? "Approving..."
            : allCategorized
              ? "✓ Approve"
              : `✓ Approve (${classifiedItems.length - selectedCategories.size} item${classifiedItems.length - selectedCategories.size !== 1 ? "s" : ""} need category)`}
        </button>
      </div>
    </div>
  );
}
```
